O que é o Sistema?

O Sistema Integrado de DP é uma aplicação web desenvolvida em React para automatizar processos do Departamento Pessoal da MAIS Escoramentos. Ele centraliza operações de folha de pagamento, gerenciamento de benefícios, geração de arquivos bancários (CNAB) e integração com o sistema ERP.

👥 Módulo: Base Local (Colaboradores)

Este módulo gerencia o cadastro de todos os colaboradores da empresa com seus dados bancários, benefícios e centro de custo. É a base fundamental do sistema - todos os outros módulos dependem destes dados.

⚠️
Importância Crítica

Sem o cadastro correto dos colaboradores nesta aba, os módulos de Salário, Benefícios e ERP não funcionarão corretamente. Sempre mantenha esta base atualizada.

Funcionalidades

  • Cadastro manual: Adicionar colaboradores um a um via formulário
  • Importação em massa: Upload de planilha Excel (.xlsx) com múltiplos colaboradores
  • Edição: Alterar dados de colaboradores existentes
  • Exclusão: Remover colaboradores (individual ou em lote)
  • Exportação: Baixar base completa em Excel
  • Busca: Filtrar por nome, CPF ou matrícula

Função de Agrupamento

JavaScript 
// Agrupa lançamentos por centro de custo
const agruparPorCentroCusto = (lancamentos, colaboradores) => {
  const resumo = {};
  
  lancamentos.forEach(lanc => {
    const colab = colaboradores.find(c => c.id === lanc.colaboradorId);
    if (!colab) return;
    
    const cc = colab.centroCusto;
    
    if (!resumo[cc]) {
      resumo[cc] = {
        centroCusto: cc,
        qtdColaboradores: 0,
        totalSalarios: 0,
        totalVT: 0,
        totalVR: 0,
        totalGeral: 0,
        colaboradores: []
      };
    }
    
    resumo[cc].qtdColaboradores++;
    resumo[cc].totalSalarios += lanc.valorLiquido || 0;
    resumo[cc].colaboradores.push(lanc);
  });
  
  // Calcula totais gerais
  Object.values(resumo).forEach(cc => {
    cc.totalGeral = cc.totalSalarios + cc.totalVT + cc.totalVR;
  });
  
  return Object.values(resumo);
};

const saveToStorage = (key, data) => {
  try {
    localStorage.setItem(key, JSON.stringify(data));

};

parseCurrency

Converte string de moeda brasileira para número.

JavaScript 
/**
 * Converte string de moeda para número
 * @param {string} value - Valor em string (ex: "1.234,56" ou "R$ 1.234,56")
 * @returns {number} Valor numérico
 */
const parseCurrency = (value) => {
  if (!value) return 0;
  
  // Remove tudo que não for dígito, vírgula ou ponto
  let cleaned = String(value).replace(/[^\d,.-]/g, '');
  
  // Substitui vírgula por ponto (formato brasileiro → JS)
  cleaned = cleaned.replace(/\./g, '').replace(',', '.');
  
  const parsed = parseFloat(cleaned);
  return isNaN(parsed) ? 0 : parsed;
};

// Exemplos:
// parseCurrency("1.234,56")     → 1234.56
// parseCurrency("R$ 1.234,56")  → 1234.56
// parseCurrency("1234")         → 1234

/**

  if (cleaned.length !== 11) return false;
  
  // Não pode ser todos dígitos iguais
  if (/^(\d)\1+$/.test(cleaned)) return false;
  
  // Validação do primeiro dígito
  let sum = 0;
  for (let i = 0; i < 9; i++) {
    sum += parseInt(cleaned[i]) * (10 - i);
  }
  let digit1 = 11 - (sum % 11);
  if (digit1 > 9) digit1 = 0;
  
  if (parseInt(cleaned[9]) !== digit1) return false;
  
  // Validação do segundo dígito
  sum = 0;
  for (let i = 0; i < 10; i++) {
    sum += parseInt(cleaned[i]) * (11 - i);
  }
  let digit2 = 11 - (sum % 11);
  if (digit2 > 9) digit2 = 0;
  
  return parseInt(cleaned[10]) === digit2;
};

const formatDate = (date, includeTime = false) => {
  if (!date) return '';

    day: '2-digit',
    month: '2-digit',

    ...(includeTime && {

  const headers = Object.keys(data[0]);
  
  // Cria linhas CSV
  const csvLines = [

    ...data.map(row => 
      headers.map(h => {
        let value = row[h];
        // Escapa valores com ponto e vírgula ou quebra de linha
        if (typeof value === 'string' && (value.includes(';') || value.includes('\n'))) {
          value = `"${value.replace(/"/g, '""')}"`;
        }
        return value ?? '';
      }).join(';')
    )
  ];
  
  const content = csvLines.join('\n');
  downloadFile(content, fileName, 'text/csv');
};

✅ Validações

Todas as validações aplicadas nos formulários e processamentos.

Validação de CPF

JavaScript 
// Regras de validação de CPF:
// 1. Deve ter exatamente 11 dígitos
// 2. Não pode ter todos os dígitos iguais (ex: 111.111.111-11)
// 3. Dígitos verificadores devem ser válidos (algoritmo Receita Federal)

const validateCPF = (cpf) => {
  const cleaned = cpf.replace(/\D/g, '');
  
  // Regra 1: 11 dígitos
  if (cleaned.length !== 11) return false;
  
  // Regra 2: não pode ser todos iguais
  if (/^(\d)\1+$/.test(cleaned)) return false;
  
  // Regra 3: validar dígitos verificadores
  let sum = 0;
  for (let i = 0; i < 9; i++) {
    sum += parseInt(cleaned[i]) * (10 - i);
  }
  let d1 = 11 - (sum % 11);
  if (d1 > 9) d1 = 0;
  if (parseInt(cleaned[9]) !== d1) return false;
  
  sum = 0;
  for (let i = 0; i < 10; i++) {
    sum += parseInt(cleaned[i]) * (11 - i);
  }
  let d2 = 11 - (sum % 11);
  if (d2 > 9) d2 = 0;
  
  return parseInt(cleaned[10]) === d2;
};

Possíveis causas:

  • LocalStorage cheio (limite ~5MB)

  • useEffect não está configurado corretamente

  2. Verifique o espaço usado
  3. Limpe dados antigos ou exporte backup

Possíveis causas:

  • PDF é imagem escaneada (não texto)
  • Worker do PDF.js não carregou

Soluções:

  1. Verifique se o PDF abre normalmente
  2. Tente selecionar texto no PDF (se não conseguir, é imagem)

  • Código do convênio inválido
  • Formato de data incorreto
  • Caracteres especiais no nome

  1. Confirme CNPJ e dados bancários com o banco
